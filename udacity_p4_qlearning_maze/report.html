import random

actions = ['u','r','d','l']
qline = {'u':1.2, 'r':-2.1, 'd':-24.5, 'l':27}
epsilon = 0.3 # 0.3

def choose_action(epsilon):
    
    action = None
    # 
    if random.random() <= epsilon:
        action = actions[random.randint(0, 3)]
    else: 
        #  Q 
        action = max(qline,key=qline.get)
#     print(action)
    return action

for i in range(1,100):
    choose_action(0.3)
